var isSorted = false;
 
while (!isSorted) {
 
  isSorted = true;
 
  for (var i = 0; i < 5 - 1; i++) {
    var left = cards[i];
    var right = cards[i+1];
 
    if (right > left) {
      var copy = left;
      cards[i] = right;
      cards[i+1] = copy;
 
      isSorted = false;
    }
 
  }
 
}
